[h1]Votes[/h1]
replaces native votes.


[h1]how to enable[/h1]
autoloaded.

[h1]Configuration options[/h1]
[h2]restartVote_enable[/h2]
Enables the feature

[b]Var Name : [/b]restartVote_enable (bool)
[h3]Default Configuration[/h3]
[code]ManiaLivePlugins\eXpansion\Votes\restartVote_enable[] = true
[/code]

[h2]restartVote_ratio[/h2]
-1 = server default, otherwise 0.00 - 1.00

[b]Var Name : [/b]restartVote_ratio (float)
[h3]Default Configuration[/h3]
[code]ManiaLivePlugins\eXpansion\Votes\restartVote_ratio[] = -1
[/code]

[h2]restartVote_timeout[/h2]
'0' for server default, '1' for indefinite, otherwise set number of desired seconds

[b]Var Name : [/b]restartVote_timeout (int)
[h3]Default Configuration[/h3]
[code]ManiaLivePlugins\eXpansion\Votes\restartVote_timeout[] = 0
[/code]

[h2]restartVote_voters[/h2]
'0' means only active players, '1' means any player, '2' is for everybody, pure spectators included

[b]Var Name : [/b]restartVote_voters (int)
[h3]Default Configuration[/h3]
[code]ManiaLivePlugins\eXpansion\Votes\restartVote_voters[] = 1
[/code]

[h2]restartVote_useQueue[/h2]
Use track queue instead of instant restart, if 'eXpansion\Maps' plugin is loaded

[b]Var Name : [/b]restartVote_useQueue (bool)
[h3]Default Configuration[/h3]
[code]ManiaLivePlugins\eXpansion\Votes\restartVote_useQueue[] = true
[/code]

[h2]skipVote_enable[/h2]
Enables the feature

[b]Var Name : [/b]skipVote_enable (bool)
[h3]Default Configuration[/h3]
[code]ManiaLivePlugins\eXpansion\Votes\skipVote_enable[] = true
[/code]

[h2]skipVote_ratio[/h2]
-1 = server default, otherwise 0.00 - 1.00

[b]Var Name : [/b]skipVote_ratio (float)
[h3]Default Configuration[/h3]
[code]ManiaLivePlugins\eXpansion\Votes\skipVote_ratio[] = -1
[/code]

[h2]skipVote_timeout[/h2]
'0' for server default, '1' for indefinite, otherwise set number of desired seconds

[b]Var Name : [/b]skipVote_timeout (int)
[h3]Default Configuration[/h3]
[code]ManiaLivePlugins\eXpansion\Votes\skipVote_timeout[] = 0
[/code]

[h2]skipVote_voters[/h2]
'0' means only active players, '1' means any player, '2' is for everybody, pure spectators included

[b]Var Name : [/b]skipVote_voters (int)
[h3]Default Configuration[/h3]
[code]ManiaLivePlugins\eXpansion\Votes\skipVote_voters[] = 1
[/code]

[h2]skipVote_limit[/h2]
Limits skip votes to within set number of seconds from beginning of match; '-1' or false to disable, set to desired # of seconds

[b]Var Name : [/b]skipVote_limit (int)
[h3]Default Configuration[/h3]
[code]ManiaLivePlugins\eXpansion\Votes\skipVote_limit[] = -1
[/code]
